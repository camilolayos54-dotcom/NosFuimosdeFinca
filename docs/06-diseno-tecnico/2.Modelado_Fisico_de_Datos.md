## 2. Scripts de Creación de Base de Datos (DDL)

Estos scripts están optimizados para el motor **PostgreSQL**, reflejando exactamente el ERD de la Fase 4.

```sql
-- Habilitar extensión para UUID
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 1. Tabla de Usuarios (MOD-AUTH)
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email VARCHAR(255) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(50) NOT NULL CHECK (role IN ('TURISTA', 'PROPIETARIO', 'ADMIN')),
    full_name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. Tabla de Fincas (MOD-FCNT)
CREATE TABLE fincas (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    owner_id UUID NOT NULL,
    title VARCHAR(150) NOT NULL,
    description TEXT,
    price_per_night DECIMAL(10, 2) NOT NULL,
    location_city VARCHAR(100) NOT NULL,
    max_guests INT NOT NULL,
    status VARCHAR(50) DEFAULT 'PENDING' CHECK (status IN ('PENDING', 'APPROVED', 'REJECTED')),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (owner_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 3. Tabla de Amenidades (MOD-FCNT)
CREATE TABLE amenities (
    id SERIAL PRIMARY KEY,
    finca_id UUID NOT NULL,
    name VARCHAR(50) NOT NULL, -- ej. 'Piscina', 'BBQ'
    FOREIGN KEY (finca_id) REFERENCES fincas(id) ON DELETE CASCADE
);

-- 4. Tabla de Imágenes (MOD-FCNT)
CREATE TABLE finca_images (
    id SERIAL PRIMARY KEY,
    finca_id UUID NOT NULL,
    image_url VARCHAR(500) NOT NULL,
    is_primary BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (finca_id) REFERENCES fincas(id) ON DELETE CASCADE
);

-- 5. Tabla Transaccional de Reservas (MOD-RSV) - CRÍTICA
CREATE TABLE bookings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    finca_id UUID NOT NULL,
    turista_id UUID NOT NULL,
    check_in DATE NOT NULL,
    check_out DATE NOT NULL,
    total_price DECIMAL(10, 2) NOT NULL,
    status VARCHAR(50) NOT NULL CHECK (status IN ('SOFT_LOCKED', 'HARD_LOCKED', 'CANCELLED')),
    locked_until TIMESTAMP WITH TIME ZONE, -- Expiración del Soft-Lock
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (finca_id) REFERENCES fincas(id) ON DELETE RESTRICT,
    FOREIGN KEY (turista_id) REFERENCES users(id) ON DELETE RESTRICT
);

-- 6. Tabla de Pagos (MOD-PAY)
CREATE TABLE payments (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    booking_id UUID NOT NULL UNIQUE,
    wompi_transaction_id VARCHAR(255),
    amount DECIMAL(10, 2) NOT NULL,
    status VARCHAR(50) NOT NULL CHECK (status IN ('PENDING', 'APPROVED', 'DECLINED', 'ERROR')),
    processed_at TIMESTAMP WITH TIME ZONE,
    FOREIGN KEY (booking_id) REFERENCES bookings(id) ON DELETE RESTRICT
);
```

## 3. Scripts de Indexación y Optimización

```sql
-- Índice en Owner_ID para acelerar el Dashboard del Propietario (MOD-POWN)
CREATE INDEX idx_fincas_owner ON fincas(owner_id);

-- Índices Compuestos para el Motor de Búsqueda (MOD-SRCH)
-- Acelera querys tipo: "Fincas en Quindío para 4 personas aprobadas"
CREATE INDEX idx_fincas_search ON fincas(status, location_city, max_guests);

-- Índices en Fechas para Reservas (MOD-RSV)
-- Crucial para validar colisiones de fechas de forma rápida antes del Soft-Lock
CREATE INDEX idx_bookings_dates ON bookings(finca_id, check_in, check_out) WHERE status != 'CANCELLED';

-- Índice para el Job Asíncrono de purgado de Soft-Locks
CREATE INDEX idx_bookings_soft_locks ON bookings(status, locked_until) WHERE status = 'SOFT_LOCKED';
```
