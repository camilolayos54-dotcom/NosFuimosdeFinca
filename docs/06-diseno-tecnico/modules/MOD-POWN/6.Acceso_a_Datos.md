public interface DashboardRepository extends Repository<Booking, UUID> {
    // Query de agregación para el Dashboard del Propietario
    @Query("SELECT COUNT(b) FROM Booking b JOIN b.finca f WHERE f.ownerId = :ownerId AND b.status = 'HARD_LOCKED'")
    long countTotalConfirmedBookings(@Param("ownerId") UUID ownerId);
}
