### 3. Nivel 2: Diagrama de Contenedores

```mermaid
C4Container
    title Container Diagram - Nos Fuimos de Finca

    Person(turista, "Turista", "Navegador Web/Móvil")
    
    System_Boundary(c1, "Nos Fuimos de Finca") {
        Container(spa, "Single Page App (SSR)", "Next.js / React", "Provee las vistas B2C y B2B al navegador vía Vercel.")
        Container(api, "API Monolithic Backend", "Java / Spring Boot", "Orquesta la lógica de negocio modular. Corre en AWS.")
        ContainerDb(db, "Relational Database", "Amazon RDS PostgreSQL", "Almacena los datos estructurados. Maneja concurrencia transaccional.")
        ContainerDb(storage, "Object Storage", "Amazon S3", "Almacena imágenes crudas de las fincas.")
    }
    
    System_Ext(wompi, "Wompi")
    
    Rel(turista, spa, "Navega y clickea", "HTTPS")
    Rel(spa, api, "Consume Endpoints REST", "JSON/HTTPS")
    Rel(api, db, "Escribe y Lee datos", "JDBC/SSL")
    Rel(api, storage, "Sube fotos", "AWS SDK")
    
    Rel(spa, wompi, "Drop-in Widget checkout", "HTTPS")
    Rel(wompi, api, "Envía Webhook de resultado", "JSON/HTTPS")
```
